** Debug (DAP)
;; Usa lldb-dap como adaptador, que viene incluido en el devshell de C++.
;; envrc se encarga de que Emacs vea el binario correcto por proyecto.
#+begin_src emacs-lisp
  (use-package dap-mode
    :after eglot
    :config
    (dap-auto-configure-mode 1)
    (require 'dap-lldb)

    ;; Buscar lldb-dap en el PATH del devshell (resuelto por envrc).
    ;; nixpkgs puede llamarlo lldb-dap (>= 18) o lldb-vscode (< 18).
    (setq dap-lldb-debug-program
          (list (or (executable-find "lldb-dap")
                    (executable-find "lldb-vscode")
                    "lldb-dap")))

    ;; Template de launch para C++
    (dap-register-debug-template
     "C++ LLDB"
     (list :type "lldb"
           :request "launch"
           :name "C++ LLDB"
           :program "${workspaceFolder}/build/${fileBasenameNoExtension}"
           :cwd "${workspaceFolder}"
           :args []
           :stopOnEntry nil))

    :bind (:map dap-mode-map
                ("C-c d d" . dap-debug)
                ("C-c d b" . dap-breakpoint-toggle)
                ("C-c d c" . dap-continue)
                ("C-c d n" . dap-next)
                ("C-c d i" . dap-step-in)
                ("C-c d o" . dap-step-out)
                ("C-c d q" . dap-disconnect)
                ("C-c d r" . dap-debug-restart)
                ("C-c d e" . dap-eval-thing-at-point)))
#+end_src

** Rust (Rustic)
;; rustic es un wrapper sobre rust-mode con integración nativa a cargo y soporte
;; para eglot. Le decimos explícitamente que use eglot (ya configurado arriba)
;; en vez de su propio cliente LSP interno.
;; rust-analyzer tiene que estar en el PATH del devshell — envrc se encarga de eso.
#+begin_src emacs-lisp
  (use-package rustic
    :custom
    ;; Usar eglot en vez del cliente LSP interno de rustic
    (rustic-lsp-client 'eglot)
    ;; Formatear con rustfmt al guardar
    (rustic-format-on-save t)

    :config
    ;; FIX: usar eglot-managed-mode-hook en vez de rustic-mode-hook.
    ;;
    ;; rustic-mode-hook se dispara cuando el modo se activa, que puede ser
    ;; ANTES de que eglot establezca la conexión (eglot-sync-connect nil →
    ;; conexión async). Si eglot-inlay-hints-mode corre en ese momento intenta
    ;; mandar una request al servidor y explota con:
    ;;   "File mode specification error: jsonrpc-error No current JSON-RPC connection"
    ;;
    ;; eglot-managed-mode-hook se dispara recién cuando eglot tomó control del
    ;; buffer, así que la conexión está garantizada. El guard derived-mode-p
    ;; limita el hook solo a buffers rustic (eglot-managed-mode-hook corre en
    ;; todos los modos que eglot maneja).
    (add-hook 'eglot-managed-mode-hook
              (lambda ()
                (when (and (derived-mode-p 'rustic-mode)
                           (fboundp 'eglot-inlay-hints-mode))
                  (eglot-inlay-hints-mode 1))))

    :bind (:map rustic-mode-map
                ;; Cargo desde Emacs — sin salir del editor
                ("C-c c r" . rustic-cargo-run)
                ("C-c c b" . rustic-cargo-build)
                ("C-c c t" . rustic-cargo-test)
                ("C-c c c" . rustic-cargo-clippy)
                ("C-c c f" . rustic-cargo-fmt)
                ("C-c c a" . rustic-cargo-add)
                ;; Ver documentación del símbolo bajo el cursor
                ("C-c c d" . eldoc-doc-buffer)))
#+end_src

** Eat
;; eat (Emulate A Terminal) es un emulador de terminal moderno y rápido para Emacs.
;; Soporta xterm-256color, mouse, imágenes inline, y se integra bien con eshell.
;; envrc se encarga de que el shell vea el entorno del devshell activo.
#+begin_src emacs-lisp
  (use-package eat
    :hook
    ;; Integración con eshell: eat reemplaza el manejo de programas de terminal
    ;; (vim, htop, etc.) dentro de eshell, sin necesidad de abrir un buffer aparte
    (eshell-load . eat-eshell-mode)

    :custom
    ;; Usar el shell definido en el sistema
    (eat-shell shell-file-name)

    :bind
    ;; Abrir terminal en el directorio del proyecto actual
    ("C-c t t" . eat)
    ("C-c t p" . eat-project))
#+end_src
