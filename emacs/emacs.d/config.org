#+begin_src emacs-lisp
  (use-package rustic
    :custom
    (rustic-lsp-client 'eglot)

** General (Leader key)
;; General permite definir keybindings con SPC como leader, igual que Doom.
;; SPC es el leader en normal/visual. M-SPC funciona en insert/emacs state.
#+begin_src emacs-lisp
  (use-package general
    :after evil
    :config
    (general-evil-setup t)

    (general-create-definer user/leader
      :keymaps '(normal visual emacs)
      :prefix "SPC"
      :global-prefix "M-SPC")

    (user/leader
      ;; Buffers
      "b b" '(consult-buffer           :which-key "switch buffer")
      "b d" '(kill-this-buffer         :which-key "kill buffer")
      "b n" '(next-buffer              :which-key "next buffer")
      "b p" '(previous-buffer          :which-key "prev buffer")

      ;; Archivos
      "f f" '(find-file                :which-key "find file")
      "f r" '(consult-recent-file      :which-key "recent files")
      "f s" '(save-buffer              :which-key "save file")

      ;; Búsqueda
      "s s" '(consult-line             :which-key "buscar en buffer")
      "s i" '(consult-imenu            :which-key "imenu")
      "s p" '(projectile-ripgrep       :which-key "ripgrep en proyecto")

      ;; Proyectos
      "p p" '(projectile-switch-project :which-key "switch proyecto")
      "p f" '(projectile-find-file      :which-key "find file en proyecto")

      ;; Git
      "g g" '(magit-status             :which-key "magit")

      ;; Ventanas
      "w v" '(split-window-right       :which-key "split vertical")
      "w s" '(split-window-below       :which-key "split horizontal")
      "w d" '(delete-window            :which-key "cerrar ventana")
      "w D" '(delete-other-windows     :which-key "cerrar resto")
      "w b" '(balance-windows          :which-key "balancear")
      "w w" '(ace-window               :which-key "saltar ventana")

      ;; Workspaces (tabs)
      "TAB n"   '(tab-bar-new-tab              :which-key "nuevo tab")
      "TAB d"   '(tab-bar-close-tab            :which-key "cerrar tab")
      "TAB r"   '(tab-bar-rename-tab           :which-key "renombrar tab")
      "TAB ]"   '(tab-bar-switch-to-next-tab   :which-key "tab siguiente")
      "TAB ["   '(tab-bar-switch-to-prev-tab   :which-key "tab anterior")
      "TAB TAB" '(tab-bar-switch-to-recent-tab :which-key "tab reciente")

      ;; LSP
      "l r" '(eglot-rename             :which-key "renombrar")
      "l a" '(eglot-code-actions       :which-key "code actions")
      "l f" '(eglot-format-buffer      :which-key "formatear")
      "l s" '(consult-eglot-symbols    :which-key "símbolos")

      ;; Terminal
      "t t" '(eat                      :which-key "terminal")
      "t p" '(eat-project              :which-key "terminal proyecto")

      ;; Help
      "h f" '(helpful-callable         :which-key "describe función")
      "h v" '(helpful-variable         :which-key "describe variable")
      "h k" '(helpful-key              :which-key "describe key")

      ;; Quit
      "q q" '(save-buffers-kill-terminal :which-key "salir")))
#+end_src

** PDF
;; pdf-tools reemplaza el DocView built-in con un renderizado nativo de alta
;; calidad usando poppler. Abre automáticamente cualquier %PDF.
#+begin_src emacs-lisp
  (use-package pdf-tools
    :magic ("%PDF" . pdf-view-mode)
    :config
    (pdf-tools-install :no-query)
    :bind (:map pdf-view-mode-map
                ("j"   . pdf-view-next-line-or-next-page)
                ("k"   . pdf-view-previous-line-or-previous-page)
                ("C-+" . pdf-view-enlarge)
                ("C--" . pdf-view-shrink)))
#+end_src
