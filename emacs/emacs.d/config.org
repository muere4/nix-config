* General
** UI Básica
#+begin_src emacs-lisp
  (add-to-list 'default-frame-alist '(fullscreen . maximized))

  ;; Transparencia del frame
  (let ((opacity 100))
    (set-frame-parameter nil 'alpha-background opacity)
    (add-to-list 'default-frame-alist `(alpha-background . ,opacity)))

  (setq inhibit-startup-message t
        use-short-answers t
        frame-resize-pixelwise t)

  (tool-bar-mode -1)
  (scroll-bar-mode -1)
  (menu-bar-mode -1)
  (set-fringe-mode 12)
  (column-number-mode)
  (delete-selection-mode 1)
  (electric-pair-mode 1)
  (save-place-mode)
  (global-display-line-numbers-mode 1)
  (blink-cursor-mode -1)

  ;; Scroll más suave
  (setq scroll-conservatively 101
        mouse-wheel-scroll-amount '(1 ((shift) . 1))
        mouse-wheel-progressive-speed nil
        mouse-wheel-follow-mouse t)

  ;; Desactivar números de línea en ciertos modos
  (dolist (mode '(term-mode-hook
                  shell-mode-hook
                  eshell-mode-hook))
    (add-hook mode (lambda () (display-line-numbers-mode 0))))
#+end_src

** Colores
#+begin_src emacs-lisp
  ;; Paleta Catppuccin Mocha hardcodeada
  (setq muere/bg-0    "#11111B"   ; crust — fondo más oscuro
        muere/bg-1    "#181825"   ; mantle — fondo principal
        muere/bg-2    "#1e1e2e"   ; base — fondo más claro
        muere/acc-0   "#CBA6F7"   ; mauve — acento principal
        muere/acc-1   "#89B4FA"   ; blue — acento secundario
        muere/acc-2   "#cdd6f4")  ; text

  ;; Aplicar colores a faces de UI
  (add-hook 'after-init-hook
            (lambda ()
              (set-face-attribute 'default nil :background muere/bg-1)
              (set-face-attribute 'fringe nil :background muere/bg-1)
              (set-face-attribute 'line-number nil :background muere/bg-1)
              (set-face-attribute 'line-number-current-line nil
                                  :background muere/bg-1
                                  :foreground muere/acc-0)
              (set-face-attribute 'mode-line nil :background muere/bg-0)
              (set-face-attribute 'mode-line-inactive nil :background muere/bg-0)
              (set-face-attribute 'vertical-border nil
                                  :background muere/bg-0
                                  :foreground muere/bg-0)))

  ;; Minibuffer con fondo más oscuro
  (add-hook 'minibuffer-setup-hook
            (lambda ()
              (face-remap-add-relative 'default `(:background ,muere/bg-0))
              (face-remap-add-relative 'fringe  `(:background ,muere/bg-0 :foreground ,muere/bg-0))))
#+end_src
